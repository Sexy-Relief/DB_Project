
CREATE TABLE customer
(
	customer_id           INTEGER NOT NULL,
	customer_name         VARCHAR(20) NOT NULL,
	phone_num	VARCHAR(30) NOT NULL,
	account_num           VARCHAR(30) NULL,
	card_num              VARCHAR(30) NULL,
	card_company          VARCHAR(20) NULL,
    PRIMARY KEY (customer_id)
);

CREATE TABLE offline_order
(
	order_id              INTEGER NOT NULL,
	product_id            INTEGER NOT NULL,
	store_id              INTEGER NOT NULL,
    PRIMARY KEY (order_id)
);

CREATE TABLE offlinestore
(
	store_id              INTEGER NOT NULL,
	store_name            VARCHAR(50) NOT NULL,
	region                VARCHAR(20) NOT NULL,
	PRIMARY KEY (store_id)
);

CREATE TABLE online_order
(
	order_id              INTEGER NOT NULL,
	product_id            INTEGER NOT NULL,
	tracking_num          INTEGER NOT NULL UNIQUE,
	warehouse_id          INTEGER NOT NULL,
    PRIMARY KEY (order_id)
);

CREATE TABLE _order
(
	order_id              INTEGER NOT NULL,
	product_id            INTEGER NOT NULL,
	customer_id           INTEGER NOT NULL,
	date_year             INTEGER NOT NULL
		CONSTRAINT  date_year CHECK (date_year BETWEEN 1900 AND 2100),
	date_mon              INTEGER NOT NULL
		CONSTRAINT  date_mon CHECK (date_mon BETWEEN 1 AND 12),
	date_day              INTEGER NOT NULL
		CONSTRAINT  date_day CHECK (date_day BETWEEN 1 AND 31),
	num_of_order          INTEGER NOT NULL
		CONSTRAINT  num_of_order CHECK (num_of_order >= 0),
	order_type            VARCHAR(20) NOT NULL,
	PRIMARY KEY (order_id)
);

CREATE TABLE product
(
	product_id            INTEGER NOT NULL,
	product_name          VARCHAR(20) NOT NULL,
	_type                  VARCHAR(20) NOT NULL,
	price                 NUMERIC(10,2) NULL
		CONSTRAINT  price CHECK (price >= 0),
	manufacturer          VARCHAR(20) NULL,
	gateway               VARCHAR(20) NULL,
    PRIMARY KEY (product_id)
);

CREATE TABLE reorder
(
	order_id              INTEGER NOT NULL,
	product_id            INTEGER NOT NULL,
	warehouse_id          INTEGER NOT NULL,
	store_id              INTEGER NOT NULL,
    PRIMARY KEY (order_id)
);

CREATE TABLE shipping
(
	tracking_num          INTEGER NOT NULL,
	shipping_company      VARCHAR(20) NOT NULL,
	address               VARCHAR(50) NOT NULL,
	isarrived             INTEGER NOT NULL
		CONSTRAINT  isarrived CHECK (isarrived BETWEEN 0 AND 1),
	got_problem           INTEGER NOT NULL
		CONSTRAINT  got_problem CHECK (got_problem BETWEEN 0 AND 1),
	PRIMARY KEY (tracking_num)
);

CREATE TABLE storestock
(
	product_id            INTEGER NOT NULL,
	store_id              INTEGER NOT NULL,
	num_of_storestock        INTEGER NOT NULL
		CONSTRAINT  num_of_storestock CHECK (num_of_storestock >= 0),
	PRIMARY KEY (product_id,store_id)
);

CREATE TABLE warehouse
(
	warehouse_id          INTEGER NOT NULL,
	warehouse_name        VARCHAR(50) NOT NULL,
    PRIMARY KEY (warehouse_id)
);

CREATE TABLE warestock
(
	product_id            INTEGER NOT NULL,
	warehouse_id          INTEGER NOT NULL,
	num_of_warestock       INTEGER NOT NULL
		CONSTRAINT  num_of_warestock CHECK (num_of_warestock >= 0),
	PRIMARY KEY (product_id,warehouse_id)
);
